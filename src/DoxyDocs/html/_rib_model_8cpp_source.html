54  void ellipseNodes(tgStructure& tetra, double a, double b, double startT, double endT, size_t n)
55  {
56  double rodL = (endT - startT) / (double) n;
57 
58  for(size_t i = 0; i < n; i++)
59  {
60  double x = a * cos( startT + i * rodL);
61  double y = b * sin( startT + i * rodL);
62  // Just build it in xy, can rotate later
63  tgNode node( btVector3( x, y, 0.0));
64  tetra.addNode( node );
65  }
66  #if (0)
67  // Final Nodes for "feet"
68  double x = a * cos( startT + (n - 1) * rodL);
69  double y = b * sin( startT + (n - 1) * rodL) - 0.5;
70  tgNode node( btVector3( x, y, 0.0));
71  tetra.addNode( node );
72  #endif
73  }

99  void addSegments(tgStructure& snake, const tgStructure& tetra,
100  const tgStructure& ribs, double edge, size_t segmentCount)
101  {
102  const btVector3 offset(0, 0, -edge * 1.3);
103  const btVector3 rotAxis(0, 1.0, 0);
104  for (size_t i = 0; i < segmentCount; ++i)
105  {
106 
107  tgStructure* const t = new tgStructure(tetra);
108  // Need a new pointer for every rib
109  tgStructure* const r = new tgStructure(ribs);
110  tgStructure* const r2 = new tgStructure(ribs);
111  t->addChild(r);
112  t->addChild(r2);
113 
114  tgNodes r01 = r->getNodes();
115  tgNodes r02 = r2->getNodes();
116  tgNodes t01 = t->getNodes();
117 
118  size_t r02End = r02.getNodes().size();
119  #if (0)
120  r2->addRotation(t01[0], rotAxis, M_PI);
121  t->addPair(t01[0], r01[0], "multiMuscle connect");
122  t->addPair(t01[0], r02[0], "multiMuscle connect");
123  #else
124  t->addPair(t01[0], r01[0], "rod connect");
125  t->addPair(t01[0], r02[0], "rod connect");
126  r2->addRotation(r02[0], rotAxis, M_PI);
127  #endif
128 
129 
130  t->addTags(tgString("segment num", i + 1));
131  t->move((i + 1) * offset);
132  // Add a child to the snake
133  snake.addChild(t);

199 void RibModel::setup(tgWorld& world)
200 {
201  double v_size = 3.0;
202 
203  // Create the spinal processes
204  tgStructure vertebrae;
205  addNodes(vertebrae, v_size);
206  addPairs(vertebrae);
207 
208  // Move the first one so we can create a longer snake.
209  // Or you could move the snake at the end, up to you.
210  vertebrae.move(btVector3(0.0, 2 * v_size, v_size * m_segments));
211 
212  // Create ribs and add them to the vertebrae
213  double majorAxis = 6.0;
214  double minorAxis = 4.0;
215  double startTheta = M_PI / 2.0;
216  double endTheta = 5.0 * M_PI / 4.0;
217  size_t segs = 15;
218 
219  tgStructure ribs;
220  ellipseNodes(ribs, majorAxis, minorAxis, startTheta, endTheta, segs);
221  makePairs(ribs);
222 
223  #if (0) // Attempt at compliant rib attachments
224  ribs.move(btVector3(v_size / 3.0, 2 * v_size - minorAxis, v_size * m_segments));
225  #else
226  ribs.move(btVector3(0.0, 2 * v_size - minorAxis -.3, v_size * m_segments));
227  #endif
228  // Create our snake segments
229  tgStructure snake;
230  addSegments(snake, vertebrae, ribs, v_size, m_segments);
231 
232  snake.move(btVector3(0.0, majorAxis, 0.0));
233 
234  addMuscles(snake);
235 
236  // Create the build spec that uses tags to turn the structure into a real model
237  // Note: This needs to be high enough or things fly apart...
238  const double density = 4.2 / 300.0;
239  const double radius = 0.5;
240  const double friction = 0.5; // Default is 0.5
241  const double rollFriction = 0.5; // Default is 0.0
242  const double restitution = 0.0; // Default
243 
244  const tgRod::Config rodConfig(radius, density, friction, rollFriction, restitution);
245  tgBuildSpec spec;
246  spec.addBuilder("rod", new tgRodInfo(rodConfig));
247 
248  const bool hist = false;
249 
250  tgLinearString::Config muscleConfig(500, 5, hist);
251  spec.addBuilder("muscle", new tgLinearStringInfo(muscleConfig));
252  tgLinearString::Config muscleConfigAct(1000, 10, hist, 0, 7000, 24, 10000);
253  spec.addBuilder("muscleAct", new tgLinearStringInfo(muscleConfigAct));
254 
255  #if (0) // Compliant Rib Attachments
256  const double stiffness = 1000;
257  const double damping = .01 * stiffness;
258 
259  tgLinearString::Config muscleConfig1(stiffness, damping, -M_PI / 2.0);
260  tgLinearString::Config muscleConfig2(stiffness, damping, M_PI / 2.0);
261  tgLinearString::Config muscleConfig3(stiffness, damping, M_PI);
262  tgLinearString::Config muscleConfig4(stiffness, damping, 0);
263 
264  spec.addBuilder("multiMuscle", new tgLinearStringInfo(muscleConfig1));
265  spec.addBuilder("multiMuscle", new tgLinearStringInfo(muscleConfig2));
266  spec.addBuilder("multiMuscle", new tgLinearStringInfo(muscleConfig3));
267  spec.addBuilder("multiMuscle", new tgLinearStringInfo(muscleConfig4));
268  #endif
269  // Create your structureInfo
270  tgStructureInfo structureInfo(snake, spec);
271 
272  // Use the structureInfo to build ourselves
273  structureInfo.buildInto(*this, world);
274 
275  // We could now use tgCast::filter or similar to pull out the models (e.g. muscles)
276  // that we want to control.
277  m_allMuscles = find<tgLinearString> ("muscleAct");
278  m_allSegments = find<tgModel> ("segment");
279 
280  #if (0)
281  trace(structureInfo, *this);
282  #endif
283 
284  // Actually setup the children
285  BaseSpineModelLearning::setup(world);
286 }
