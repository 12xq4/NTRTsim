140 void BaseSpineCPGControl::onSetup(BaseSpineModelLearning& subject)
141 {
142  m_pCPGSys = new CPGEquations();
143  //Initialize the Learning Adapters
144  nodeAdapter.initialize(&nodeEvolution,
145  nodeLearning,
146  nodeConfigData);
147  edgeAdapter.initialize(&edgeEvolution,
148  edgeLearning,
149  edgeConfigData);
150  /* Empty vector signifying no state information
151  * All parameters are stateless parameters, so we can get away with
152  * only doing this once
153  */
154  std::vector<double> state;

156 
157  array_4D edgeParams = scaleEdgeActions(edgeAdapter.step(dt, state));
158  array_2D nodeParams = scaleNodeActions(nodeAdapter.step(dt, state));
159 
160  setupCPGs(subject, nodeParams, edgeParams);
161 
162  initConditions = subject.getSegmentCOM(m_config.segmentNumber);
163 #if (0) // Conditional compile for data logging
164  m_dataObserver.onSetup(subject);
165 #endif
166 
167 #if (0) // Conditional Compile for debug info
168  std::cout << *m_pCPGSys << std::endl;
169 #endif
170  m_updateTime = 0.0;

173 void BaseSpineCPGControl::setupCPGs(BaseSpineModelLearning& subject, array_2D nodeActions, array_4D edgeActions)
174 {
175 
176  std::vector <tgLinearString*> allMuscles = subject.getAllMuscles();
177 
178  for (std::size_t i = 0; i < allMuscles.size(); i++)
179  {
180  tgCPGStringControl* pStringControl = new tgCPGStringControl();
181  allMuscles[i]->attach(pStringControl);
182  m_allControllers.push_back(pStringControl);
183  }
184 
186  // First assign node numbers to the info Classes
187  for (std::size_t i = 0; i < m_allControllers.size(); i++)
188  {
189  m_allControllers[i]->assignNodeNumber(*m_pCPGSys, nodeActions);
190  }
191 
192  // Then determine connectivity and setup string
193  for (std::size_t i = 0; i < m_allControllers.size(); i++)
194  {
195  tgCPGStringControl * const pStringInfo = m_allControllers[i];
196  assert(pStringInfo != NULL);
197  pStringInfo->setConnectivity(m_allControllers, edgeActions);
198 
199  //String will own this pointer
200  ImpedanceControl* p_ipc = new ImpedanceControl( m_config.tension,
201  m_config.kPosition,
202  m_config.kVelocity);
203  if (m_config.useDefault)
204  {
205  pStringInfo->setupControl(*p_ipc);
206  }
207  else
208  {
209  pStringInfo->setupControl(*p_ipc, m_config.controlLength);
210  }
211  }

215 void BaseSpineCPGControl::onStep(BaseSpineModelLearning& subject, double dt)
216 {
217  m_updateTime += dt;
218  if (m_updateTime >= m_config.controlTime)
219  {
220  std::size_t numControllers = subject.getNumberofMuslces();
221 
222  double descendingCommand = 2.0;
223  std::vector<double> desComs (numControllers, descendingCommand);
224 
225  m_pCPGSys->update(desComs, m_updateTime);
226 #if (0) // Conditional compile for data logging
227  m_dataObserver.onStep(subject, m_updateTime);
228 #endif
229  m_updateTime = 0;

233 void BaseSpineCPGControl::onTeardown(BaseSpineModelLearning& subject)
234 {
235  std::vector<double> scores;
236  // @todo - check to make sure we ran for the right amount of time
237 
238  std::vector<double> finalConditions = subject.getSegmentCOM(m_config.segmentNumber);
239 
240  const double newX = finalConditions[0];
241  const double newZ = finalConditions[2];
242  const double oldX = initConditions[0];
243  const double oldZ = initConditions[2];
244 
245  const double distanceMoved = sqrt((newX-oldX) * (newX-oldX) +
246  (newZ-oldZ) * (newZ-oldZ));
247 
248  scores.push_back(distanceMoved);
249 
252  double totalEnergySpent=0;
253 
254  vector<tgLinearString* > tmpStrings = subject.getAllMuscles();
255  for(int i=0; i<tmpStrings.size(); i++)
256  {
257  tgBaseString::BaseStringHistory stringHist = tmpStrings[i]->getHistory();
258 
259  for(int j=1; j<stringHist.tensionHistory.size(); j++)
260  {
261  const double previousTension = stringHist.tensionHistory[j-1];
262  const double previousLength = stringHist.restLengths[j-1];
263  const double currentLength = stringHist.restLengths[j];
264  //TODO: examine this assumption - free spinning motor may require more power
265  double motorSpeed = (currentLength-previousLength);
266  if(motorSpeed > 0) // Vestigial code
267  motorSpeed = 0;
268  const double workDone = previousTension * motorSpeed;
269  totalEnergySpent += workDone;
270  }
271  }
272 
273  scores.push_back(totalEnergySpent);
274 
275  edgeAdapter.endEpisode(scores);
276  nodeAdapter.endEpisode(scores);
277 
278  delete m_pCPGSys;
279  m_pCPGSys = NULL;
280 
281  for(size_t i = 0; i < m_allControllers.size(); i++)
282  {
283  delete m_allControllers[i];
284  }
285  m_allControllers.clear();
286 }

288 array_4D BaseSpineCPGControl::scaleEdgeActions
289  (vector< vector <double> > actions)
290 {
291  std::size_t numControllers = edgeConfigData.getintvalue("numberOfControllers");
292 
293  // Ensure reading from the same file
294  assert(numControllers == actions.size());
295  assert(actions[0].size() == 2);
296 
297  double lowerLimit = m_config.lowPhase;
298  double upperLimit = m_config.highPhase;
299  double range = upperLimit - lowerLimit;
300 
301  array_4D actionList(boost::extents[m_config.segmentSpan]
302  [m_config.theirMuscles]
303  [m_config.ourMuscles]
304  [m_config.params]);
305 
306  /* Horrid while loop to populate upper diagonal of matrix, since
307  * its symmetric and we want to minimze parameters used in learing
308  * note that i==1, j==k will refer to the same muscle
309  * @todo use boost to set up array so storage is only allocated for
310  * elements that are used
311  */
312  int i = 0;
313  int j = 0;
314  int k = 0;
315 
316  while (i < m_config.segmentSpan)
317  {
318  while(j < m_config.theirMuscles)
319  {
320  while(k < m_config.ourMuscles)
321  {
322  if (actions.empty())
323  {
324  std::cout << "ran out before table populated!"
325  << std::endl;
326  break;
327  }
328  else
329  {
330  if (i == 1 && j == k)
331  {
332  // std::cout << "Skipped identical muscle" << std::endl;
333  //Skip since its the same muscle
334  }
335  else
336  {
337  std::vector<double> edgeParam = actions.back();
338  // Weight from 0 to 1
339  actionList[i][j][k][0] = edgeParam[0];
340  // Phase offset from -pi to pi
341  actionList[i][j][k][1] = edgeParam[1] *
342  (range) + lowerLimit;
343  actions.pop_back();
344  }
345  }
346  k++;
347  }
348  j++;
349  k = j;
350 
351  }
352  j = 0;
353  k = 0;
354  i++;
355  }
356 
357  assert(actions.empty());
358 
359  return actionList;
360 }
361 array_2D BaseSpineCPGControl::scaleNodeActions
362  (vector< vector <double> > actions)
363 {
364  std::size_t numControllers = nodeConfigData.getintvalue("numberOfControllers");
365  std::size_t numActions = nodeConfigData.getintvalue("numberOfActions");
366 
367  assert( actions.size() == numControllers);
368  assert( actions[0].size() == numActions);
369 
370  array_2D nodeActions(boost::extents[numControllers][numActions]);
371 
372  array_2D limits(boost::extents[2][numActions]);
373 
374  // Check if we need to update limits
375  assert(numActions == 2);
376 
377 
378  for (int i = 0; i !=2; i++)
379  {
380 
381  limits[0][i] = m_config.lowAmp;
382  limits[1][i] = m_config.highAmp;
383  }
384 
385  // This one is square
386  for( std::size_t i = 0; i < numControllers; i++)
387  {
388  for( std::size_t j = 0; j < numActions; j++)
389  {
390  nodeActions[i][j] = ( actions[i][j] *
391  (limits[1][j] - limits[0][j])) + limits[0][j];
392  }
393  }
394 
395  return nodeActions;
