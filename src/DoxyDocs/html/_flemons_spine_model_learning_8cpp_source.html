46 void FlemonsSpineModelLearning::setup(tgWorld& world)
47 {
48  // This is basically a manual setup of a model.
49  // There are things that do this for us
51 
52  // Rod and Muscle configuration
53  // Note: This needs to be high enough or things fly apart...
54  const double density = 4.2/300.0;
55  const double radius = 0.5;
56  const tgRod::Config rodConfig(radius, density);
57 
58  tgLinearString::Config muscleConfig(1000, 10, false, 0, 7000, 12, 4000);
59 
60  // Calculations for the flemons spine model
61  double v_size = 10.0;
62 
63  // Create the tetrahedra
64  tgStructure tetra;
65 
66  tetra.addNode(0.0, 0.0, 0.0); // center
67  tetra.addNode( v_size, v_size, v_size); // front
68  tetra.addNode( v_size, -v_size, -v_size); // right
69  tetra.addNode(-v_size, v_size, -v_size); // back
70  tetra.addNode(-v_size, -v_size, v_size); // left
71 
72  tetra.addPair(0, 1, "front rod");
73  tetra.addPair(0, 2, "right rod");
74  tetra.addPair(0, 3, "back rod");
75  tetra.addPair(0, 4, "left rod");
76 
77  // Move the first one so we can create a longer snake.
78  // Or you could move the snake at the end, up to you.
79  tetra.move(btVector3(0.0,15.0,100.0));
80 
81  // Create our snake segments
82  tgStructure snake;
85  btVector3 offset(0.0, 0.0, -v_size * 1.15);
86  for (std::size_t i = 0; i < m_segments; i++)
87  {
91  tgStructure* const p = new tgStructure(tetra);
92  p->addTags(tgString("segment num", i + 1));
93  p->move((i + 1.0) * offset);
94  snake.addChild(p); // Add a child to the snake
95  }
96  //conditionally compile for debugging
97 #if (1)
98  // Add muscles that connect the segments
99  // Tag the muscles with their segment numbers so CPGs can find
100  // them.
101  std::vector<tgStructure*> children = snake.getChildren();
102  for (std::size_t i = 1; i < children.size(); i++)
103  {
104  tgNodes n0 = children[i - 1]->getNodes();
105  tgNodes n1 = children[i]->getNodes();
106 
107  snake.addPair(n0[1], n1[1],
108  tgString("outer front muscle seg", i - 1) + tgString(" seg", i));
109  snake.addPair(n0[2], n1[2],
110  tgString("outer right muscle seg", i - 1) + tgString(" seg", i));
111  snake.addPair(n0[3], n1[3],
112  tgString("outer back muscle seg", i - 1) + tgString(" seg", i));
113  snake.addPair(n0[4], n1[4],
114  tgString("outer top muscle seg", i - 1) + tgString(" seg", i));
115 
116  snake.addPair(n0[2], n1[1],
117  tgString("inner front muscle seg", i - 1) + tgString(" seg", i));
118  snake.addPair(n0[2], n1[4],
119  tgString("inner right muscle seg", i - 1) + tgString(" seg", i));
120  snake.addPair(n0[3], n1[1],
121  tgString("inner left muscle seg", i - 1) + tgString(" seg", i));
122  snake.addPair(n0[3], n1[4],
123  tgString("inner back muscle seg", i - 1) + tgString(" seg", i));
124 
125  }
126 #endif
127  // Create the build spec that uses tags to turn the structure into a real model
128  tgBuildSpec spec;
129  spec.addBuilder("rod", new tgRodInfo(rodConfig));
130 
131 #if (1)
132  spec.addBuilder("muscle", new tgLinearStringInfo(muscleConfig));
133 #endif
134 
135  // Create your structureInfo
136  tgStructureInfo structureInfo(snake, spec);
137 
138  // Use the structureInfo to build ourselves
139  structureInfo.buildInto(*this, world);
140 
141  // Setup vectors for control
142  m_allMuscles = tgCast::filter<tgModel, tgLinearString> (getDescendants());
143 
144  m_allSegments = this->find<tgModel> ("segment");
145 
146 #if (0)
147  // Debug printing
148  std::cout << "StructureInfo:" << std::endl;
149  std::cout << structureInfo << std::endl;
150 
151  std::cout << "Model: " << std::endl;
152  std::cout << *this << std::endl;
153 #endif
154  children.clear();
155 
156  // Actually setup the children, notify controller
157  BaseSpineModelLearning::setup(world);
158 }
