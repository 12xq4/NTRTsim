229 void SpiralSpine::setup(tgWorld& world)
230 {
231  // This is basically a manual setup of a model.
232  // There are things that do this for us.
233  // (@todo: reference the things that do this for us)
234 
235  const tgRod::Config rodConfig(c.radius, c.density);
236  tgLinearString::Config muscleConfig(c.stiffness, c.damping);
237 
238  tgStructure s;
239  addNodes(s);
240  addRods(s);
241  addMuscles(s);
242  s.move(btVector3(0, 10, 0));
243 
244  // Create the build spec that uses tags to turn the structure into a real model
245  tgBuildSpec spec;
246  spec.addBuilder("rod", new tgRodInfo(rodConfig));
247  spec.addBuilder("muscle", new tgLinearStringInfo(muscleConfig));
248 
249  // Create your structureInfo
250  tgStructureInfo structureInfo(s, spec);
251 
252  // Use the structureInfo to build ourselves
253  structureInfo.buildInto(*this, world);
254 
255  // We could now use tgCast::filter or similar to pull out the
256  // models (e.g. muscles) that we want to control.
257  allMuscles = tgCast::filter<tgModel, tgLinearString> (getDescendants());
258  mapMuscles(muscleMap, *this);
259 
260  std::cout << structureInfo << std::endl;
261 
262  // Actually setup the children
263  tgModel::setup(world);
264 }
