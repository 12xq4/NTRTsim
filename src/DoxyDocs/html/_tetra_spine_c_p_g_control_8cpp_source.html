48 void TetraSpineCPGControl::setupCPGs(BaseSpineModelLearning& subject, array_2D nodeActions, array_4D edgeActions)
49 {
50  std::vector <tgLinearString*> allMuscles = subject.getAllMuscles();
51 
52  for (std::size_t i = 0; i < allMuscles.size(); i++)
53  {
54  tgCPGStringControl* pStringControl = new tgCPGStringControl();
55  allMuscles[i]->attach(pStringControl);
56  m_allControllers.push_back(pStringControl);
57  }
58 
60  // First assign node numbers to the info Classes
61  for (std::size_t i = 0; i < m_allControllers.size(); i++)
62  {
63  m_allControllers[i]->assignNodeNumber(*m_pCPGSys, nodeActions);
64  }
65 
66  double tension;
67  double kPosition;
68  double kVelocity;
69  double controlLength;
70  // Then determine connectivity and setup string
71  for (std::size_t i = 0; i < m_allControllers.size(); i++)
72  {
73  tgCPGStringControl * const pStringInfo = m_allControllers[i];
74  assert(pStringInfo != NULL);
75  pStringInfo->setConnectivity(m_allControllers, edgeActions);
76 
77  //String will own this pointer
78 #if (0) // origninal params
79  if (allMuscles[i]->hasTag("outer"))
80  {
81  tension = 0.0;
82  kPosition = 1000.0;
83  kVelocity = 100.0;
84  controlLength = 17.0;
85  }
86  else
87  {
88  tension = 0.0;
89  kPosition = 1000.0;
90  kVelocity = 100.0;
91  controlLength = 15.0 ;
92  }
93 #else // Params for In Won
94  if (allMuscles[i]->hasTag("outer"))
95  {
96  tension = 0.0;
97  kPosition = 2000.0;
98  kVelocity = 300.0;
99  controlLength = 17.5;
100  }
101  else
102  {
103  tension = 0.0;
104  kPosition = 2000.0;
105  kVelocity = 300.0;
106  controlLength = 15.0 ;
107  }
108 #endif
109  ImpedanceControl* p_ipc = new ImpedanceControl( tension,
110  kPosition,
111  kVelocity);
112  pStringInfo->setupControl(*p_ipc, controlLength);
113  }
114 
115 }
